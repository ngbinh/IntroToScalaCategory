Functor

  
/*
 * A structure-preserving map between two algebraic structures.
 */
trait Functor[F[_]] {
  def map[A, B](fa: F[A])(f: A => B): F[B]
}
  

  Functors compose! Given any functor F[_] and any functor G[_] we can create a new functor F[G[_]] by composing them:
  
val listOpt = Functor[List] compose Functor[Option]
listOpt.map(List(Some(1), None, Some(3)))(_ + 1)
List(Some(2), None, Some(4))
