Natural transformation

A way of transforming one functor into another while respecting the internal structure (i.e. the composition of morphisms) of the categories involved

  
trait NaturalTransformation[F[_], G[_]] { self =>
  def apply[A](fa: F[A]): G[A]

  def compose[E[_]](f: NaturalTransformation[E, F])
  : NaturalTransformation[E, G] =
    new NaturalTransformation[E, G] {
      def apply[A](fa: E[A]): G[A] = self.apply(f(fa))
    }
}
