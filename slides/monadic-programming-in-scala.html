Monadic programming in Scala

  
val lengthOption = convertNumericStringToInt(m.group(3).trim)
// => Option[Int]
val timeUnitOption = durationMap.get(m.group(4).trim)
// => Option[String]

for {
    i <- lengthOption
    tu <- timeUnitOption
} yield FiniteDuration(i, tu)  // => Option[FiniteDuration]
